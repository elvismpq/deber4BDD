match (p:Persona)-[r:Conoce]->(l:Lugar) where l.nombre="Parque Nacional Cotopaxi" return p;

match (p:Persona)-[r:RELATED_TO]->(pe:Pelicula) where p.nombre="Elvis Pérez" return pe;
